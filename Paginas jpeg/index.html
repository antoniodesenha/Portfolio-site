<script>

  // CONFIGURAÇÕES

  const totalPages = 48;          // total de páginas

  const pdfPages = [];            // ex: [3, 10] se quiser trocar page3.jpg por page3.pdf

  const prefix = "page";          // prefixo dos arquivos

  const ext = "jpg";              // extensão usada (.jpg)


  const container = document.getElementById('content');


  function makePage(i) {

    const wrapper = document.createElement('div');

    wrapper.className = 'page';


    if (pdfPages.includes(i)) {

      const iframe = document.createElement('iframe');

      iframe.src = `${prefix}${i}.${'pdf'}`;

      iframe.loading = 'lazy';

      wrapper.appendChild(iframe);

    } else {

      const img = document.createElement('img');

      img.src = `${prefix}${i}.${ext}`;

      img.alt = `Página ${i}`;

      img.loading = 'lazy';

      wrapper.appendChild(img);

    }

    return wrapper;

  }


  for (let i = 1; i <= totalPages; i++) {

    container.appendChild(makePage(i));

  }

</script>
